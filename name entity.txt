import spacy

nlp = spacy.load("en_core_web_sm")
text="Barack Obama was the 44th President of the United States. He was born in hawii"
doc = nlp(text)
entities = [(ent.text, ent.label) for ent in doc.ents]
print("Named Entities:")
print(entities)
